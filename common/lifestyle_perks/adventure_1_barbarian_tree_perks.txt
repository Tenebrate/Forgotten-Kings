
### ADVENTURE - BARBARIAN PERKS


# Rage
rage_perk = {
	lifestyle = adventure_lifestyle
	tree = barbarian
	position = { 1 0 }
	icon = node_martial
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_trait = beserker
			}
			add = 1989
		}
		if = {
			limit = {
				has_trait = wroth
			}
			add = 3
		}
		if = {
			limit = {
				has_focus = adventure_barbarian_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}
	
	effect = {
		add_trait_force_tooltip = aggressive_attacker
		custom_description_no_bullet = {
			text = rage_perk_effect
		}
	}
}

	auto_selection_weight = {
		value = 10
		if = {
			limit = {
				has_education_stewardship_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = stewardship_domain_focus
			}

		}
	}
	
	parent = rage_perk

	character_modifier = {
		holding_build_gold_cost = -0.05

		build_gold_cost = -0.05
		build_piety_cost = -0.05
		build_prestige_cost = -0.05
	}
}
